#include<iostream>
using namespace std;
int main()
{
	int n;
	cin >> n;
	int i;
	//int arr[7] = { 1, 4, 0, 0, 3, 10, 5};
	int arr[n];
	for(i=0;i<n;i++)
	{
	    cin>>arr[i];
	}
	
	int sum;
	cin >> sum;
	int count = 0;
	int j;
	
	for (i = 0;i < n;i++)
	{
		for (j = i;j < n;j++)
		{
		    if (arr[i] + arr[j] == sum)
		    {
			   count = 1;
			   cout << "sum found in index " << i << " and " << j << endl;
			   break;
		    }
	    }
	}
	if (count == 0)
	{
		cout << "not found" << endl;
	}
}
